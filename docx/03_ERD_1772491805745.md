# مخطط الكيانات والعلاقات (ERD) - نظام "ليالي العُمر"

يمثل الـ Diagram التالي البنية التحتية لقاعدة البيانات (Database Schema)، مع مراعاة كافة العلاقات التي تدعم الـ (RBAC)، سجلات التدقيق، وسير الحجوزات.

```mermaid
erDiagram
    BRANCH {
        int id PK
        string name "اسم الفرع"
        string location "الموقع"
        boolean is_active "لتطبيق Soft Delete"
    }

    ROLE {
        int id PK
        string name "System Administrator, Branch Manager, Employee"
        string permissions "JSON array of granted permissions (CREATE_INVOICE, VIEW_REPORTS, etc)"
    }

    USER {
        int id PK
        int role_id FK "دور المستخدم في النظام"
        int branch_id FK "الفرع التابع له (لـ OBAC)"
        string full_name "اسم الموظف"
        string phone
        string password_hash
        boolean is_active
    }

    DRESS {
        int id PK
        int current_branch_id FK "الفرع الذي يحتوي على الفستان"
        string name "اسم الفستان / الموديل"
        string dress_type "نوع الفستان"
        string current_status "متاح، محجوز، مصنع، صيانة، نقل"
        string chest_size "مقاس الصدر"
        string waist_size "مقاس الخصر"
        int max_usage_limit "العدد الأقصى للبسات المسموح بها"
        int current_usage_count "عدد اللبسات الفعلية"
        boolean is_active
    }

    BOOKING_INVOICE {
        int id PK
        int dress_id FK
        int created_by FK "الموظف المنشئ للفاتورة"
        string contract_type "Sale, Rent (إيجار/بيع)"
        string customer_name "اسم العميل"
        string customer_phone "جوال العميل"
        date reserved_for_date "تاريخ الزفاف"
        date output_date "تاريخ استلام العميل للفستان"
        date return_date "تاريخ إعادة الفستان"
        string dress_adjustments "تعديلات للفستان"
        string accessories "التوابع"
        decimal total_amount "إجمالي المبلغ"
        decimal paid_advance "المبلغ المقدم"
        decimal remaining_amount "المبلغ المتبقي"
        string status "Active, Completed, Cancelled"
        datetime created_at
    }

    RECEIPT {
        int id PK
        int booking_invoice_id FK "الفاتورة"
        int received_by FK "المحصل"
        decimal amount "قيمة الدفعة المستلمة"
        string receipt_nature "Deposit (عربون), Partial (جزء), Final (تصفية)"
        datetime created_at
    }

    EXPENSE {
        int id PK
        int branch_id FK
        int logged_by FK
        decimal amount
        string description "وصف المصروف"
        datetime created_at
    }

    TRANSFER_REQUEST {
        int id PK
        int dress_id FK
        int from_branch_id FK
        int to_branch_id FK
        int requested_by FK
        int approved_by FK
        string status "Pending, Approved, InTransit, Received, Rejected"
        datetime created_at
    }

    AUDIT_LOG {
        int id PK
        int user_id FK "من الفاعل؟"
        string action_type "CREATE, UPDATE, DELETE (Soft)"
        string table_name "اسم الجدول المتأثر"
        int record_id "المعرف الخاص بالسطر المتأثر"
        string old_values "JSON للبيانات قبل التعديل"
        string new_values "JSON للبيانات بعد التعديل"
        datetime created_at
    }
    
    SYSTEM_INSIGHT {
        int id PK
        string title "عنوان التحليل"
        text content "محتوى اقتراح Gemini API"
        string level "INFO, WARNING, CRITICAL"
        datetime created_at
    }

    %% Relationships
    ROLE ||--o{ USER : "assigns permissions"
    BRANCH ||--o{ USER : "employs (OBAC Scope)"
    BRANCH ||--o{ DRESS : "stores"
    BRANCH ||--o{ EXPENSE : "incurs"
    USER ||--o{ BOOKING_INVOICE : "creates"
    USER ||--o{ RECEIPT : "issues"
    USER ||--o{ EXPENSE : "logs"
    DRESS ||--o{ BOOKING_INVOICE : "reserved via"
    BOOKING_INVOICE ||--o{ RECEIPT : "paid through"
    USER ||--o{ AUDIT_LOG : "triggers actions"
    BRANCH ||--o{ TRANSFER_REQUEST : "sends/receives"
    USER ||--o{ TRANSFER_REQUEST : "requests/approves"
    DRESS ||--o{ TRANSFER_REQUEST : "transfers"
```
